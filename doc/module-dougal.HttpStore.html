Type

Example
store.url(new Car()); // /cars
store.url(new Car({id: 123})); // /cars/123
// works with all attributes:
partsStore.url(new CarPart({carId: 123, id: 456}));
// /cars/123/parts/456 => baseUrl: /cars/{{carId}}/parts/{{id}}
